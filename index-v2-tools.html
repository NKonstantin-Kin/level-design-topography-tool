// Новый класс для элементов class DesignElement { constructor(type, x, y, color) { this.type = type; // 'line', 'rect', 'text' this.x = x; this.y = y; this.color = color || '#000000'; } } // Инициализация инструментов function initTools() { const tools = [ { name: 'Line', icon: '✏️', action: startDrawingLine }, { name: 'Rect', icon: '⬛', action: startDrawingRect } ]; tools.forEach(tool => { const btn = document.createElement('button'); btn.innerHTML = `${tool.icon} ${tool.name}`; btn.onclick = tool.action; document.getElementById('toolbar').appendChild(btn); }); }
